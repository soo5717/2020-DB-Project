CREATE TABLE SUBJECT(
	subject_id number NOT NULL,
	subject_name varchar(100) NOT NULL,
	subject_grade number NOT NULL,
	subject_credit number NOT NULL,
	field_id number NOT NULL,
	type_id number NOT NULL,
	department_id number NOT NULL,
	CONSTRAINT subject_pk PRIMARY KEY(subject_id),
	CONSTRAINT subject_fk1 FOREIGN KEY (field_id) REFERENCES FIELD(field_id),
	CONSTRAINT subject_fk2 FOREIGN KEY (type_id) REFERENCES TYPES(type_id),
	CONSTRAINT subject_fk3 FOREIGN KEY (department_id) REFERENCES DEPARTMENTS(department_id)
)
